138template <class T>
139void MutablePriorityQueue<T>::heapifyUp(unsigned i) {
140 auto x = H[i];
141 while (i > 1 && *x < *H[parent(i)]) {
142 set(i, H[parent(i)]);
143 i = parent(i);
144 }
145 set(i, x);
146}

148template <class T>
149void MutablePriorityQueue<T>::heapifyDown(unsigned i) {
150 auto x = H[i];
151 while (true) {
152 unsigned k = leftChild(i);
153 if (k >= H.size())
154 break;
155 if (k+1 < H.size() && *H[k+1] < *H[k])
156 ++k; // right child of i
157 if ( ! (*H[k] < *x) )
158 break;
159 set(i, H[k]);
160 i = k;
161 }
162 set(i, x);
163}
